install raspberry pi os
apt install libgpiod-dev
git clone https://github.com/gshearer/sump_monitor
edit sump_notify.sh and update to your liking
apt install libgpiod-dev
mkdir /usr/local/share/sound
mv assets/sound.wav /usr/local/share/sound
gcc -Wall -s -o sump_monitord sump_monitord.c -lgpiod
mv sump_monitord sump_notify.sh /usr/local/bin
mv sump_monitor.service /etc/systemd/system
chown root:root /usr/local/bin/sump_notify.sh /usr/local/bin/sump_monitord /etc/systemd/system/sump_monitor.service
chmod 0700 /usr/local/bin/sump_notify.sh
chmod 0500 /usr/local/bin/sump_monitord
chmod 0644 /etc/systemd/system/sump_monitor.service
systemctl enable --now sump_monitor.service
test!
